const SHEET_NAME = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { const params = e.parameter || {}; const body = e.postData ? JSON.parse(e.postData.contents) : {}; const action = params.action || body.action; const sheet = SHEET_NAME; ensureSheets(sheet); if (action === 'read') { const tab = params.tab || body.tab; const s = sheet.getSheetByName(tab); const data = s.getDataRange().getValues(); if (data.length === 0) return ok({ rows: [] }); const headers = data[0]; const rows = data.slice(1).map(r => { let obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return ok({ rows }); } if (action === 'write') { const tab = body.tab; const rows = body.rows; const s = sheet.getSheetByName(tab); const headers = s.getRange(1, 1, 1, s.getLastColumn()).getValues()[0]; rows.forEach(row => { const vals = headers.map(h => row[h] !== undefined ? row[h] : ''); s.appendRow(vals); }); return ok({ written: rows.length }); } if (action === 'updateRow') { const tab = body.tab; const idField = body.idField; const idValue = body.idValue; const updates = body.updates; const s = sheet.getSheetByName(tab); const data = s.getDataRange().getValues(); const headers = data[0]; const idCol = headers.indexOf(idField); for (let i = 1; i < data.length; i++) { if (String(data[i][idCol]) === String(idValue)) { Object.entries(updates).forEach(([k, v]) => { const col = headers.indexOf(k); if (col >= 0) s.getRange(i+1, col+1).setValue(v); }); break; } } return ok({ updated: true }); } if (action === 'deleteRow') { const tab = body.tab; const idField = body.idField; const idValue = body.idValue; const s = sheet.getSheetByName(tab); const data = s.getDataRange().getValues(); const headers = data[0]; const idCol = headers.indexOf(idField); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][idCol]) === String(idValue)) { s.deleteRow(i+1); break; } } return ok({ deleted: true }); } return ok({ error: 'Accion desconocida' }); } catch (err) { return ok({ error: err.toString() }); } } function ensureSheets(ss) { const tabs = { 'Productos': ['id','barcode','name','category','price','cost','stock','unit'], 'Ventas': ['id','date','total','subtotal','discount','paymentMethod','amountReceived','change','customerId','customerName','sellerName','items'], 'VentaItems': ['saleId','productId','productName','qty','price','subtotal'], 'Clientes': ['id','name','phone','email','notes','totalPurchases','totalAmount','createdAt'], 'Usuarios': ['id', 'name', 'pin', 'role', 'active'] }; Object.entries(tabs).forEach(([name, headers]) => { let s = ss.getSheetByName(name); if (!s) { s = ss.insertSheet(name); s.appendRow(headers); } }); } function ok(data) { return ContentService.createTextOutput(JSON.stringify({ ok: true, ...data })).setMimeType(ContentService.MimeType.JSON); }
